function createSpeedCheck(attributes) {
    'use strict';
    // Private members
    attributes = attributes || {};
    attributes.speed = attributes.speed || 0;
    attributes.licencePlate = attributes.licencePlate || '???';
    attributes.infraction = false;
    attributes.speedLimit = attributes.speedLimit || 130;
    attributes.platesPattern = attributes.platesPattern || /[A-Z]{2}\d{3}[A-Z]{2}/;

    // Private functions
    function validateInfraction(speed) {
        attributes.infraction = (speed > attributes.speedLimit) ? true : false;
    }

    function validatePlate(plate) {
            return (plate.match(attributes.platesPattern) !== null);
        }
        // create and return the new Object with  public functions / getters / setter
    return {
        toString: function() {
            return 'Véhicule ' + attributes.licencePlate +
                ' roule à ' + attributes.speed + ' km/h. ' +
                (attributes.infraction ? 'Infraction!' : 'Ça va, circulez...');
        },
        isInfraction: function() {
            return attributes.infraction;
        },
        getSpeed: function() {
            return attributes.speed;
        },
        setSpeed: function(speed) {
            if (speed < 0) {
                throw {
                    name: 'SpeedCheckError',
                    message: 'Vitesse négative. Vous roulez a contre sens?'
                };
            }
            validateInfraction(speed);
            attributes.speed = speed;
        },
        getLicencePlate: function() {
            return attributes.licencePlate;
        },
        setLicencePlate: function(licencePlate) {
            if (!validatePlate(licencePlate)) {
                throw {
                    name: 'SpeedCheckError',
                    message: 'Plaque non reconnue. Vous avez de la chance.'
                };
            }
            attributes.licencePlate = licencePlate;
        }
    };
}

Partie 3 - Base de données

Question 11)

Le terme "noSQL" signifie "not only SQL". Il désigne un ensemble relativement nouveau de systèmes gestions de bases de données qui ne correspondent plus aux propriétés et architectures classiques des basses de données. En particulier les notions de tables, de tuples et de jointures deviennent obsolètes dans ces systèmes. On notera que le terme est très mal choisi car il fait penser a "no SQL" ("pas de SQL"), or ce n'est pas forcement vrai, voir complètement faux dans certaines bases.

Question 12)

MongoDB vs. MySQL, ou plus généralement, quand utiliser une base noSQL et quand utiliser une base classique ? Tout dépend de l'application visée. Les bases de données classiques excellent quand il s'agit de mettre en relation des données et maintenir la cohérences de ces relations a tout prix. Les bases noSQL relâchent les contraintes d'atomicité et de cohérence au profit de bien meilleurs temps d'accès en lecture et écriture. Les bases noSQL sont plus adaptés au stockage a gros volume et grosse fréquence de mise ajours pour des données avec pas ou peu d’interconnexions en elles. la facilité de distribution et de réplication est aussi un atout des bases noSQL.
